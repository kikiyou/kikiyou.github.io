循环语句 for：

  • 第一种形式：
    func main() {
    a := 1
    for {
    a++
if a > 3 {
    break
}
    }
    fmt.Println(a)
    }
  • 第二种形式
    func main(){
    a := 1
    for a <= 3 {
    a++
    }
    fmt.Println(a)
    }
  • 第三种形式
    func main() {
    a := 1
    for i := 0; i < 3; i++ {
    a++
    }
    fmt.Println(a)
    }
